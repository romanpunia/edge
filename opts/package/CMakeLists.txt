#Resolve platform for needed generator
if (DEFINED ENV{VCPKG_ROOT} AND NOT DEFINED CMAKE_TOOLCHAIN_FILE)
    if (NOT CMAKE_GENERATOR_PLATFORM)
        if (CMAKE_SIZEOF_VOID_P EQUAL 8)
            if (WIN32)
                set(VCPKG_TARGET_TRIPLET "x64-windows")
            elseif (UNIX)
                set(VCPKG_TARGET_TRIPLET "x64-linux")
            elseif (${CMAKE_SYSTEM_NAME} MATCHES "Darwin")
                set(VCPKG_TARGET_TRIPLET "x64-osx")
            endif()
            set(_VCPKG_TARGET_TRIPLET_ARCH "x64")
        else()
            if (WIN32)
                set(VCPKG_TARGET_TRIPLET "x86-windows")
            elseif (UNIX)
                set(VCPKG_TARGET_TRIPLET "x86-linux")
            elseif (${CMAKE_SYSTEM_NAME} MATCHES "Darwin")
                set(VCPKG_TARGET_TRIPLET "x86-osx")
            endif()
            set(_VCPKG_TARGET_TRIPLET_ARCH "x86")
        endif()
    endif()
    set(CMAKE_TOOLCHAIN_FILE "$ENV{VCPKG_ROOT}/scripts/buildsystems/vcpkg.cmake" CACHE STRING "")
endif()

#Resource management system
function(add_resource_batch DIRECTORY OUTPUT)
    set(BUFFER "#ifndef HAS_RESOURCE_BRANCH\n#define HAS_RESOURCE_BRANCH\n\nnamespace resource_batch\n{")
    file(GLOB_RECURSE BINARIES ${DIRECTORY}/*)
    foreach(BINARY ${BINARIES})
        string(REPLACE "${DIRECTORY}" "" FILENAME ${BINARY})
        string(REPLACE "${DIRECTORY}/" "" FILENAME ${BINARY})
        string(REGEX REPLACE "\\.| |-" "_" FILENAME ${FILENAME})
        string(REPLACE "/" "_" FILENAME ${FILENAME})
        string(TOUPPER ${FILENAME} DEFINENAME)
        string(TOLOWER ${FILENAME} FILENAME)
        file(READ ${BINARY} FILEDATA HEX)
        if (NOT FILEDATA STREQUAL "")
            string(REGEX REPLACE "([0-9a-f][0-9a-f])" "0x\\1," FILEDATA ${FILEDATA})
            string(APPEND BUFFER "\n\tnamespace ${FILENAME}\n\t{\n#define HAS_${DEFINENAME}\n\t\tconst unsigned char data[] = { ${FILEDATA} }\;\n\t\tconst unsigned size = sizeof(data)\;\n\t}\n")
            message(STATUS "Resource ${FILENAME} found")
        endif()
    endforeach()
    string(APPEND BUFFER "}\n#endif")
    file(WRITE ${OUTPUT} ${BUFFER})
    message(STATUS "Resource files have been written to: ${OUTPUT}")
endfunction()
